<script type="text/javascript">
document.addEventListener("DOMContentLoaded", () => {
  const isSP = window.innerWidth <= 767;
  const slideshow = isSP
    ? document.querySelector(".sp-slideshow")
    : document.querySelector(".pc-slideshow");

  if (!slideshow) return;

  const slides = slideshow.querySelectorAll(".top-slide");
  const overlay = document.querySelector(".top-overlay");
  const fadeDuration = 2000;
  const displayDuration = 7000;
  let current = 0;

  // 初期化
  slides.forEach(slide => {
    slide.style.backgroundImage = `url('${slide.dataset.bg}')`;
    slide.style.opacity = 0;
    slide.style.transform = "scale(1.2)";
    slide.style.position = "absolute";
    slide.style.top = 0;
    slide.style.left = 0;
    slide.style.width = "100%";
    slide.style.height = "100%";
    slide.style.transition = `opacity ${fadeDuration}ms ease-in-out, transform ${displayDuration + fadeDuration*2}ms linear`;
    slide.style.zIndex = 1;
  });

  // 最初の画像読み込み
  const firstImage = new Image();
  firstImage.src = slides[0].dataset.bg;
  firstImage.onload = () => {
    setTimeout(() => {
      slides[current].style.opacity = 1;
      slides[current].style.transform = "scale(1)";
      overlay.style.background = "rgba(0,0,0,0.35)";
    }, 500);
  };

  // クロスフェード関数
  function crossFade(nextIndex) {
    const prev = slides[current];
    const next = slides[nextIndex];

    next.style.opacity = 0;
    next.style.transform = "scale(1.2)";
    next.style.zIndex = 2;

    setTimeout(() => {
      next.style.opacity = 1;
      next.style.transform = "scale(1)";
    }, 50);

    prev.style.opacity = 0;
    prev.style.transform = "scale(1)";
    prev.style.zIndex = 1;

    current = nextIndex;
    setTimeout(() => crossFade((current + 1) % slides.length), displayDuration);
  }

  // 自動切替スタート
  setTimeout(() => crossFade((current + 1) % slides.length), displayDuration);

  // リサイズ時はスライド切替は続ける
  window.addEventListener("resize", () => {
    // 特に処理せず、そのまま自動切替
  });
});
</script>
